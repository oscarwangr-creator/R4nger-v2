import logging

class BaseModule:
    """
    Base class for all R4nger modules, providing lifecycle management,
    logging, error handling, and state management.
    """

    def __init__(self):
        self.state = 'initialized'
        self.logger = self.setup_logging()
        self.logger.info('BaseModule initialized')

    def setup_logging(self):
        """
        Sets up logging for the module.
        """
        logger = logging.getLogger(self.__class__.__name__)
        handler = logging.StreamHandler()
        formatter = logging.Formatter('%(asctime)s - %(name)s - %(levelname)s - %(message)s')
        handler.setFormatter(formatter)
        logger.addHandler(handler)
        logger.setLevel(logging.INFO)
        return logger

    def start(self):
        """
        Start the module lifecycle.
        """
        try:
            self.state = 'running'
            self.logger.info('Module started')
            self.run()
        except Exception as e:
            self.logger.error(f'Error in starting module: {e}')
            self.state = 'error'

    def run(self):
        """
        Main logic for the module (to be implemented in subclasses).
        """
        raise NotImplementedError('Run method must be implemented by subclasses')

    def stop(self):
        """
        Stop the module lifecycle.
        """
        self.logger.info('Stopping module')
        self.state = 'stopped'

    def get_state(self):
        """
        Get current state of the module.
        """
        return self.state

import subprocess
import json
import logging

from typing import List, Dict, Optional, Set

from abc import ABC, abstractmethod

logger = logging.getLogger(__name__)

class BaseTool(ABC):
    """Base class for all tools"""
    def __init__(self):
        self.name = ""
        self.description = ""
    
    @abstractmethod
    def execute(self, **kwargs) -> Dict:
        pass

# OSINT TOOLS - 35+ implementations

class AmmassTool(BaseTool):
    def __init__(self):
        self.name = "amass"
        self.description = "OWASP Amass - DNS enumeration and subdomain discovery"
    
    def execute(self, domain: str, **kwargs) -> Dict:
        try:
            cmd = ['amass', 'enum', '-d', domain, '-json', '-']
            result = subprocess.run(cmd, capture_output=True, text=True, timeout=1800)
            subdomains = set()
            for line in result.stdout.strip().split('\n'):
                if line:
                    try:
                        data = json.loads(line)
                        if data.get('name'):
                            subdomains.add(data['name'])
                    except: pass
            return {'status': 'success', 'subdomains': list(subdomains), 'count': len(subdomains)}
        except Exception as e:
            return {'status': 'error', 'error': str(e)}

#!/usr/bin/env python3
"""
R4nger-V3 Pipeline Engine - YAML-based workflow orchestration
Supports multi-stage execution, fault tolerance, and intelligent looping
"""

import yaml
import json
import logging
import asyncio
from typing import Dict, List, Any, Optional
from datetime import datetime
from pathlib import Path
import importlib

logger = logging.getLogger(__name__)

class PipelineEngine:
    """YAML pipeline executor with fault tolerance"""
    
    def __init__(self):
        self.pipelines = {}
        self.results = {}
        self.tools = {}
        self.logger = logging.getLogger(__name__)
    
    def load_pipeline(self, pipeline_path: str) -> Dict:
        """Load YAML pipeline definition"""
        try:
            with open(pipeline_path, 'r') as f:
                pipeline = yaml.safe_load(f)
            return pipeline
        except Exception as e:
            self.logger.error(f"Failed to load pipeline: {e}")
            return {}
    
    def register_tool(self, tool_name: str, tool_class):
        """Register a tool for use in pipelines"""
        self.tools[tool_name] = tool_class()
    
    async def execute_stage(self, stage: Dict, context: Dict) -> Dict:
        """Execute a single pipeline stage"""
        try:
            stage_name = stage.get('name', 'unknown')
            tool_name = stage.get('tool')
            params = stage.get('params', {})
            
            # Resolve context variables in params
            resolved_params = self._resolve_context(params, context)
            
            if tool_name not in self.tools:
                return {'status': 'error', 'error': f"Tool {tool_name} not found"}
            
            tool = self.tools[tool_name]
            result = tool.execute(**resolved_params)
            
            # Store result in context
            context[f"stage_{stage_name}"] = result
            
            return {
                'status': 'success',
                'stage': stage_name,
                'result': result,
                'timestamp': datetime.now().isoformat()
            }
        except Exception as e:
            return {'status': 'error', 'error': str(e), 'stage': stage.get('name', 'unknown')}
    
    def _resolve_context(self, params: Dict, context: Dict) -> Dict:
        """Replace $VAR references with context values"""
        resolved = {}
        for key, value in params.items():
            if isinstance(value, str) and value.startswith('$'):
                var_name = value[1:]
                resolved[key] = context.get(var_name, value)
            else:
                resolved[key] = value
        return resolved
    
    async def execute_pipeline(self, pipeline: Dict) -> Dict:
        """Execute complete pipeline"""
        try:
            pipeline_name = pipeline.get('name', 'unnamed')
            stages = pipeline.get('stages', [])
            context = pipeline.get('context', {})
            
            results = {
                'pipeline': pipeline_name,
                'started_at': datetime.now().isoformat(),
                'stages': [],
                'status': 'running'
            }
            
            for stage in stages:
                if stage.get('enabled', True) is False:
                    continue
                
                stage_result = await self.execute_stage(stage, context)
                results['stages'].append(stage_result)
                
                # Check for failure handling
                if stage_result.get('status') == 'error':
                    on_error = stage.get('on_error', 'continue')
                    if on_error == 'abort':
                        results['status'] = 'aborted'
                        break
                    elif on_error == 'retry':
                        # Retry logic
                        retries = stage.get('retries', 3)
                        for retry in range(retries):
                            stage_result = await self.execute_stage(stage, context)
                            if stage_result.get('status') == 'success':
                                results['stages'][-1] = stage_result
                                break
            
            results['completed_at'] = datetime.now().isoformat()
            results['status'] = 'completed'
            
            return results
        except Exception as e:
            return {
                'pipeline': pipeline.get('name', 'unnamed'),
                'status': 'failed',
                'error': str(e)
            }
    
    def validate_pipeline(self, pipeline: Dict) -> tuple:
        """Validate pipeline structure"""
        errors = []
        
        if 'name' not in pipeline:
            errors.append("Missing 'name' field")
        
        if 'stages' not in pipeline or not isinstance(pipeline['stages'], list):
            errors.append("Missing or invalid 'stages' field")
        
        for i, stage in enumerate(pipeline.get('stages', [])):
            if 'tool' not in stage:
                errors.append(f"Stage {i} missing 'tool' field")
        
        return len(errors) == 0, errors

// Tab switching
function switchTab(tabName) {
    // Hide all tabs
    document.querySelectorAll('.tab-content').forEach(tab => {
        tab.classList.remove('active');
    });
    
    // Remove active class from nav items
    document.querySelectorAll('.nav-item').forEach(item => {
        item.classList.remove('active');
    });
    
    // Show selected tab
    document.getElementById(tabName).classList.add('active');
    
    // Find and highlight nav item
    event.target.classList.add('active');
    
    state.currentTab = tabName;
}

async function loadTools() {

function renderToolsList() {

            <button onclick="selectTool('${tool.name}')" class="btn btn-primary btn-small">Use Tool</button>
        `;
        container.appendChild(card);
    });
}

function selectTool(toolName) {
    state.selectedTool = toolName;
    showToolDetail(toolName);
}

async function showToolDetail(toolName) {
    try {
        const response = await fetch(`${API_BASE}/tools/${toolName}`);
        const tool = await response.json();
        
        const panel = document.getElementById('tool-detail-panel');
        document.getElementById('tool-name').textContent = tool.name;
        document.getElementById('tool-description').textContent = tool.description;
        
        // Generate parameter form
        const paramsDiv = document.getElementById('tool-params');
        paramsDiv.innerHTML = `
            <div class="form-group">
                <label>Target/Query:</label>
                <input type="text" id="tool-param-target" placeholder="Enter target domain, IP, or URL" />
            </div>
            <div class="form-group">
                <label>Additional Parameters (JSON):</label>
                <textarea id="tool-param-extra" placeholder='{"key": "value"}' style="min-height:100px;"></textarea>
            </div>
        `;
        
        panel.style.display = 'block';
    } catch (error) {
        showToast('Failed to load tool', 'error');
    }
}

async function executeTool() {
    const toolName = state.selectedTool;
    const target = document.getElementById('tool-param-target').value;
    const extra = document.getElementById('tool-param-extra').value;
    
    if (!target) {
        showToast('Please enter a target', 'warning');
        return;
    }
    
    const params = { target, ...JSON.parse(extra || '{}') };
    
    try {
        const response = await fetch(`${API_BASE}/tools/${toolName}/execute`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(params)
        });
        
        const result = await response.json();
        showToast('Tool execution started', 'success');
        
        // Poll for results
        pollTaskResult(result.task_id);
    } catch (error) {
        showToast('Failed to execute tool', 'error');
    }
}

function filterTools() {

    cards.forEach(card => {

        if (!filter || category === filter) {
            card.style.display = 'block';
        } else {
            card.style.display = 'none';
        }
    });
}

function renderTasksList() {

    tbody.innerHTML = '';

        const row = tbody.insertRow();

        const startedAt = new Date(task.started_at).toLocaleString();

            </td>
        `;
    });

async function pollTaskResult(taskId, maxRetries = 60) {
    for (let i = 0; i < maxRetries; i++) {
        try {
            const response = await fetch(`${API_BASE}/tasks/${taskId}`);
            const task = await response.json();
            
            if (task.status === 'completed' || task.status === 'failed') {
                showTaskResult(taskId, task);
                await loadTasks();
                return;
            }
        } catch (error) {
            console.error('Poll error:', error);
        }
        
        await new Promise(r => setTimeout(r, 1000));
    }
}

function showTaskResult(taskId, task) {
    const panel = document.getElementById('tool-output');
    const output = document.getElementById('tool-output-text');
    
    output.textContent = JSON.stringify(task.result, null, 2);

        showToast('Task completed successfully', 'success');
    } else {

    }
}
